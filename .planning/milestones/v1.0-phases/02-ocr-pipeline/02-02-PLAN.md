```typescript
import OpenAI from 'openai'
import { NextResponse } from 'next/server'

// Read OPENAI_API_KEY from server-only env. NEVER prefix with NEXT_PUBLIC_.
// (T-2-01 mitigation — see 02-RESEARCH.md Security Domain.)
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY })

const RECEIPT_PROMPT = `You are a receipt parser. Extract every line item and its price from this receipt image.
Return ONLY valid JSON matching this schema exactly:
{ "items": [{ "name": string, "priceCents": number }] }
Rules:
- priceCents must be an integer (e.g. $12.99 -> 1299)
- name should be a short readable description (3-6 words max)
- Exclude subtotals, tax, tip, and total lines
- If you cannot read an item clearly, include your best guess`

// Vercel Hobby tier allows up to 60s; 30s is generous for gpt-4o-mini vision
// on a ~500KB receipt image while keeping client-side overlay UX bounded.
export const maxDuration = 30

export async function POST(request: Request) {
  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }

  const image =
    body && typeof body === 'object' && 'image' in body
      ? (body as { image: unknown }).image
      : undefined

  if (typeof image !== 'string' || image.length === 0) {
    return NextResponse.json({ error: 'No image provided' }, { status: 400 })
  }

  try {
    const completion = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [
        {
          role: 'user',
          content: [
            { type: 'text', text: RECEIPT_PROMPT },
            { type: 'image_url', image_url: { url: image, detail: 'high' } },
          ],
        },
      ],
      response_format: {
        type: 'json_schema',
        json_schema: {
          name: 'receipt_items',
          strict: true,
          schema: {
            type: 'object',
            properties: {
              items: {
                type: 'array',
                items: {
                  type: 'object',
                  properties: {
                    name: { type: 'string' },
                    priceCents: { type: 'integer' },
                  },
                  required: ['name', 'priceCents'],
                  additionalProperties: false,
                },
              },
            },
            required: ['items'],
            additionalProperties: false,
          },
        },
      },
    })

    const content = completion.choices[0]?.message?.content
    if (!content) {
      console.error('OCR error: empty response from gpt-4o-mini')
      return NextResponse.json({ error: 'OCR failed' }, { status: 500 })
    }

    const parsed = JSON.parse(content) as {
      items: { name: string; priceCents: number }[]
    }

    return NextResponse.json(parsed)
  } catch (err) {
    // Log server-side only. Do NOT echo OpenAI internals to the client.
    console.error('OCR error:', err)
    return NextResponse.json({ error: 'OCR failed' }, { status: 500 })
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser → /api/ocr | Untrusted base64 string crosses here; route validates type and presence before any OpenAI call |
| /api/ocr → OpenAI API | Server-to-server; OPENAI_API_KEY authenticates this leg only |
| /api/ocr → browser response | Server-controlled JSON; OpenAI internals must NOT be reflected back |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-2-01 | Information Disclosure | OPENAI_API_KEY | mitigate | Read via `process.env.OPENAI_API_KEY` at module scope only; acceptance criterion `grep -c NEXT_PUBLIC_OPENAI` returns 0 in route.ts |
| T-2-02 | Tampering | Malformed/missing image field in request body | mitigate | Body parsed in try/catch; type-check `image` is a non-empty string; return 400 before any OpenAI call |
| T-2-03 | Tampering | Malformed JSON content from OpenAI | mitigate | `JSON.parse(content)` inside try/catch; null/empty content path; both return generic 500 with `{ error: 'OCR failed' }` |
| T-2-04 | Information Disclosure | OpenAI error details leaking into response body | mitigate | Catch block logs `err` to `console.error` only; response body is the literal string `{ error: 'OCR failed' }`; test asserts the thrown message ("rate-limited xyz123") is NOT in the response JSON |
| T-2-05 | Denial of Service | Oversized request body | mitigate (defense-in-depth) | Browser-image-compression caps at ~500KB before POST (Plan 03); route's `maxDuration = 30` bounds compute time; further size limits are within Vercel's default 4.5MB function payload — no extra config needed at this scale |
| T-2-06 | Tampering | Prompt injection via image text | mitigate | `response_format` json_schema with `strict: true` constrains output to the items schema regardless of injected text; `additionalProperties: false` rejects extras |
| T-2-07 | Repudiation | OCR failure with no log trace | accept | `console.error('OCR error:', err)` provides server-side log trail; full audit logging is out of scope for v1 |
</threat_model>
