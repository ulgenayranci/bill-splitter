```typescript
import { NextResponse } from 'next/server'
import { redis } from '@/lib/redis'
import type { SessionPayload } from '@/lib/sessionSchema'

export const maxDuration = 10

export async function POST(
  request: Request,
  { params }: { params: Promise<{ sessionId: string }> }
) {
  const { sessionId } = await params

  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }
  const b = body && typeof body === 'object' ? (body as Record<string, unknown>) : {}
  const personId = b.personId
  const tipCents = b.tipCents

  if (typeof personId !== 'string' || personId.length === 0) {
    return NextResponse.json({ error: 'Invalid personId' }, { status: 400 })
  }
  if (typeof tipCents !== 'number' || !Number.isInteger(tipCents) || tipCents < 0) {
    return NextResponse.json({ error: 'Invalid tipCents: must be integer >= 0' }, { status: 400 })
  }

  try {
    const session = await redis.get<SessionPayload>(`session:${sessionId}`)
    if (!session) {
      return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    }
    if (!session.people.some((p) => p.id === personId)) {
      return NextResponse.json({ error: 'Invalid personId: not in session' }, { status: 400 })
    }

    const updated: SessionPayload = {
      ...session,
      tips: { ...(session.tips ?? {}), [personId]: tipCents },
    }
    await redis.set(`session:${sessionId}`, JSON.stringify(updated), { ex: 86400 })
    return NextResponse.json({ ok: true })
  } catch (err) {
    console.error('Tip error:', err)
    return NextResponse.json({ error: 'Tip failed' }, { status: 500 })
  }
}
```

```typescript
import { nanoid } from 'nanoid'
import { NextResponse } from 'next/server'
import { redis } from '@/lib/redis'
import type { SessionPayload, EditRequest, EditRequestPayload } from '@/lib/sessionSchema'

export const maxDuration = 10

const VALID_TYPES = ['add', 'remove', 'edit_price', 'edit_name'] as const
type EditType = (typeof VALID_TYPES)[number]

function validatePayload(
  type: EditType,
  payload: unknown,
  session: SessionPayload
): { ok: true; payload: EditRequestPayload } | { ok: false; error: string } {
  if (!payload || typeof payload !== 'object') return { ok: false, error: 'Invalid payload' }
  const p = payload as Record<string, unknown>

  if (type === 'add') {
    if (typeof p.name !== 'string' || p.name.length === 0) return { ok: false, error: 'Invalid add payload: name' }
    if (!Number.isInteger(p.priceCents) || (p.priceCents as number) <= 0) return { ok: false, error: 'Invalid add payload: priceCents' }
    if (!Number.isInteger(p.quantity) || (p.quantity as number) <= 0) return { ok: false, error: 'Invalid add payload: quantity' }
    return { ok: true, payload: { type: 'add', name: p.name, priceCents: p.priceCents as number, quantity: p.quantity as number } }
  }

  // remove / edit_price / edit_name all require itemId that exists in session.items
  if (typeof p.itemId !== 'string' || p.itemId.length === 0) return { ok: false, error: 'Invalid payload: itemId' }
  if (!session.items.some((it) => it.id === p.itemId)) return { ok: false, error: 'Invalid payload: itemId not in session' }

  if (type === 'remove') return { ok: true, payload: { type: 'remove', itemId: p.itemId as string } }
  if (type === 'edit_price') {
    if (!Number.isInteger(p.newPriceCents) || (p.newPriceCents as number) <= 0) return { ok: false, error: 'Invalid edit_price payload' }
    return { ok: true, payload: { type: 'edit_price', itemId: p.itemId as string, newPriceCents: p.newPriceCents as number } }
  }
  // edit_name
  if (typeof p.newName !== 'string' || p.newName.length === 0) return { ok: false, error: 'Invalid edit_name payload' }
  return { ok: true, payload: { type: 'edit_name', itemId: p.itemId as string, newName: p.newName } }
}

export async function POST(
  request: Request,
  { params }: { params: Promise<{ sessionId: string }> }
) {
  const { sessionId } = await params

  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }
  const b = body && typeof body === 'object' ? (body as Record<string, unknown>) : {}
  const personId = b.personId
  const type = b.type

  if (typeof personId !== 'string' || personId.length === 0) {
    return NextResponse.json({ error: 'Invalid personId' }, { status: 400 })
  }
  if (typeof type !== 'string' || !(VALID_TYPES as readonly string[]).includes(type)) {
    return NextResponse.json({ error: 'Invalid type' }, { status: 400 })
  }

  try {
    const session = await redis.get<SessionPayload>(`session:${sessionId}`)
    if (!session) {
      return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    }
    if (!session.people.some((p) => p.id === personId)) {
      return NextResponse.json({ error: 'Invalid personId: not in session' }, { status: 400 })
    }
    const payloadResult = validatePayload(type as EditType, b.payload, session)
    if (!payloadResult.ok) {
      return NextResponse.json({ error: payloadResult.error }, { status: 400 })
    }

    const requestId = nanoid()
    const editRequest: EditRequest = {
      personId,
      type: type as EditType,
      payload: payloadResult.payload,
      status: 'pending',
      createdAt: Date.now(),
    }
    const updated: SessionPayload = {
      ...session,
      editRequests: { ...(session.editRequests ?? {}), [requestId]: editRequest },
    }
    await redis.set(`session:${sessionId}`, JSON.stringify(updated), { ex: 86400 })
    return NextResponse.json({ ok: true, requestId })
  } catch (err) {
    console.error('Edit request error:', err)
    return NextResponse.json({ error: 'Edit request failed' }, { status: 500 })
  }
}
```

### File A: `app/api/session/[sessionId]/resolve-edit/route.ts`

This route applies the requested item mutation atomically. Because the mutation is a compound update (multiple fields, sometimes including claims deletion for remove), use `redis.eval` with a Lua script for the approve path. The reject path is a simple `redis.set` (single field update).

```typescript
import { NextResponse } from 'next/server'
import { redis } from '@/lib/redis'
import type { SessionPayload } from '@/lib/sessionSchema'

export const maxDuration = 10

/**
 * Lua: approve an edit request. Atomically:
 *  - sets editRequests[requestId].status = 'approved'
 *  - applies item mutation based on type:
 *    - add: appends new item to session.items with quantity
 *    - remove: deletes item from session.items AND deletes claims.items[itemId]
 *    - edit_price: updates item.priceCents
 *    - edit_name: updates item.name
 * Returns 'OK' | 'session_not_found' | 'request_not_found' | 'wrong_status' | 'invalid_session'
 */
const APPROVE_EDIT_SCRIPT = `
local raw = redis.call('GET', KEYS[1])
if not raw then return 'session_not_found' end
local ok, session = pcall(cjson.decode, raw)
if not ok then return 'invalid_session' end

local requestId = ARGV[1]
local newItemId = ARGV[2]  -- only for 'add'; empty otherwise

if not session.editRequests or not session.editRequests[requestId] then
  return 'request_not_found'
end
local req = session.editRequests[requestId]
if req.status ~= 'pending' then return 'wrong_status' end

req.status = 'approved'

local p = req.payload
if req.type == 'add' then
  if not session.items then session.items = {} end
  table.insert(session.items, {
    id = newItemId,
    name = p.name,
    priceCents = p.priceCents,
    quantity = p.quantity,
  })
elseif req.type == 'remove' then
  if session.items then
    local kept = {}
    for _, it in ipairs(session.items) do
      if it.id ~= p.itemId then table.insert(kept, it) end
    end
    session.items = kept
  end
  if session.claims and session.claims.items then
    session.claims.items[p.itemId] = nil
  end
elseif req.type == 'edit_price' then
  if session.items then
    for _, it in ipairs(session.items) do
      if it.id == p.itemId then it.priceCents = p.newPriceCents end
    end
  end
elseif req.type == 'edit_name' then
  if session.items then
    for _, it in ipairs(session.items) do
      if it.id == p.itemId then it.name = p.newName end
    end
  end
end

redis.call('SET', KEYS[1], cjson.encode(session), 'EX', 86400)
return 'OK'
`

export async function POST(
  request: Request,
  { params }: { params: Promise<{ sessionId: string }> }
) {
  const { sessionId } = await params

  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }
  const b = body && typeof body === 'object' ? (body as Record<string, unknown>) : {}
  const hostToken = b.hostToken
  const requestId = b.requestId
  const decision = b.decision

  if (typeof hostToken !== 'string' || hostToken.length === 0) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }
  if (typeof requestId !== 'string' || requestId.length === 0) {
    return NextResponse.json({ error: 'Invalid requestId' }, { status: 400 })
  }
  if (decision !== 'approved' && decision !== 'rejected') {
    return NextResponse.json({ error: 'Invalid decision' }, { status: 400 })
  }

  try {
    const session = await redis.get<SessionPayload>(`session:${sessionId}`)
    if (!session) {
      return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    }
    if (session.hostToken !== hostToken) {
      return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
    }
    if (!session.editRequests?.[requestId]) {
      return NextResponse.json({ error: 'request_not_found' }, { status: 404 })
    }

    if (decision === 'rejected') {
      // Simple update: just change status. No item mutation. Single-field set is fine.
      const updated: SessionPayload = {
        ...session,
        editRequests: {
          ...session.editRequests,
          [requestId]: { ...session.editRequests[requestId], status: 'rejected' },
        },
      }
      await redis.set(`session:${sessionId}`, JSON.stringify(updated), { ex: 86400 })
      return NextResponse.json({ ok: true })
    }

    // Approved path uses Lua for atomic compound mutation.
    // For 'add' we need a new item id pre-generated server-side.
    const { nanoid } = await import('nanoid')
    const newItemId = nanoid()
    const result = await redis.eval(
      APPROVE_EDIT_SCRIPT,
      [`session:${sessionId}`],
      [requestId, newItemId]
    )
    if (result === 'session_not_found') return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    if (result === 'request_not_found') return NextResponse.json({ error: 'request_not_found' }, { status: 404 })
    if (result === 'wrong_status') return NextResponse.json({ error: 'request already resolved' }, { status: 409 })
    if (result === 'invalid_session') return NextResponse.json({ error: 'invalid_session' }, { status: 500 })

    return NextResponse.json({ ok: true })
  } catch (err) {
    console.error('Resolve edit error:', err)
    return NextResponse.json({ error: 'Resolve edit failed' }, { status: 500 })
  }
}
```

```typescript
import { NextResponse } from 'next/server'
import { redis } from '@/lib/redis'
import type { SessionPayload } from '@/lib/sessionSchema'

export const maxDuration = 10

/**
 * Lua: resolve a dispute. Atomically:
 *  - if decision='resolved' and reassignTo is non-empty: sets disputes[disputeId].status='resolved',
 *    deletes claims.items[itemId][originalPersonId], adds claims.items[itemId][reassignTo] = { qty=1, assignedBy='host' }
 *  - if decision='resolved' and reassignTo is empty: just sets status='resolved' (host confirmed assignment as-is)
 *  - if decision='rejected': sets status='rejected', no claims mutation
 * Returns 'OK' | 'session_not_found' | 'dispute_not_found' | 'invalid_session' | 'wrong_status'
 */
const RESOLVE_DISPUTE_SCRIPT = `
local raw = redis.call('GET', KEYS[1])
if not raw then return 'session_not_found' end
local ok, session = pcall(cjson.decode, raw)
if not ok then return 'invalid_session' end

local disputeId = ARGV[1]
local decision = ARGV[2]      -- 'resolved' | 'rejected'
local reassignTo = ARGV[3]    -- personId or empty string

if not session.disputes or not session.disputes[disputeId] then
  return 'dispute_not_found'
end
local d = session.disputes[disputeId]
if d.status ~= 'pending' then return 'wrong_status' end

d.status = decision

if decision == 'resolved' and reassignTo ~= '' then
  if not session.claims then session.claims = {} end
  if not session.claims.items then session.claims.items = {} end
  if not session.claims.items[d.itemId] then session.claims.items[d.itemId] = {} end
  -- Remove the original disputed claim, add the host reassignment
  session.claims.items[d.itemId][d.personId] = nil
  session.claims.items[d.itemId][reassignTo] = { qty = 1, assignedBy = 'host' }
end

redis.call('SET', KEYS[1], cjson.encode(session), 'EX', 86400)
return 'OK'
`

export async function POST(
  request: Request,
  { params }: { params: Promise<{ sessionId: string }> }
) {
  const { sessionId } = await params

  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }
  const b = body && typeof body === 'object' ? (body as Record<string, unknown>) : {}
  const hostToken = b.hostToken
  const disputeId = b.disputeId
  const decision = b.decision
  const reassignTo = b.reassignTo

  if (typeof hostToken !== 'string' || hostToken.length === 0) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }
  if (typeof disputeId !== 'string' || disputeId.length === 0) {
    return NextResponse.json({ error: 'Invalid disputeId' }, { status: 400 })
  }
  if (decision !== 'resolved' && decision !== 'rejected') {
    return NextResponse.json({ error: 'Invalid decision' }, { status: 400 })
  }
  if (reassignTo !== undefined && (typeof reassignTo !== 'string' || reassignTo.length === 0)) {
    return NextResponse.json({ error: 'Invalid reassignTo' }, { status: 400 })
  }

  try {
    const session = await redis.get<SessionPayload>(`session:${sessionId}`)
    if (!session) {
      return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    }
    if (session.hostToken !== hostToken) {
      return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
    }
    if (!session.disputes?.[disputeId]) {
      return NextResponse.json({ error: 'dispute_not_found' }, { status: 404 })
    }
    // Validate reassignTo references a real person if supplied
    if (reassignTo && !session.people.some((p) => p.id === reassignTo)) {
      return NextResponse.json({ error: 'Invalid reassignTo: not in session' }, { status: 400 })
    }

    const result = await redis.eval(
      RESOLVE_DISPUTE_SCRIPT,
      [`session:${sessionId}`],
      [disputeId, decision, reassignTo ?? '']
    )
    if (result === 'session_not_found') return NextResponse.json({ error: 'session_not_found' }, { status: 404 })
    if (result === 'dispute_not_found') return NextResponse.json({ error: 'dispute_not_found' }, { status: 404 })
    if (result === 'wrong_status') return NextResponse.json({ error: 'dispute already resolved' }, { status: 409 })
    if (result === 'invalid_session') return NextResponse.json({ error: 'invalid_session' }, { status: 500 })

    return NextResponse.json({ ok: true })
  } catch (err) {
    console.error('Resolve dispute error:', err)
    return NextResponse.json({ error: 'Resolve dispute failed' }, { status: 500 })
  }
}
```

CRITICAL:
- HostToken validation is TWO-step: (1) presence/type check returns 403 BEFORE touching Redis (saves a roundtrip); (2) session-loaded hostToken comparison returns 403 if mismatched. Never trust client.
- The `APPROVE_EDIT_SCRIPT` accepts a pre-generated `newItemId` from JS — Lua cannot easily generate IDs, so JS generates via nanoid and passes through. This is safe because the ID is server-generated.
- The `RESOLVE_DISPUTE_SCRIPT` uses `reassignTo` empty-string sentinel (Lua has no native nil/empty distinction over the wire reliably).
- `wrong_status` returns 409 Conflict — preserves idempotency semantics (already-approved request can't be re-approved).
- The `remove` edit type deletes `session.claims.items[itemId]` in the same atomic op (RESEARCH Open Question 3).
